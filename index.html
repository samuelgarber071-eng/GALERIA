// Estado global da aplicação
let appState = {
    folders: ['Minhas Fotos', 'Viagens', 'Trabalho'],
    currentFolder: 'Minhas Fotos',
    isTrashView: false,
    photos: [] 
};

// Carrega os dados persistidos no LocalStorage
if(localStorage.getItem('photo_os_state')) {
    try {
        const saved = JSON.parse(localStorage.getItem('photo_os_state'));
        appState = saved;
        if (!appState.folders.includes(appState.currentFolder)) {
            appState.currentFolder = appState.folders[0] || '';
        }
        appState.isTrashView = false;
    } catch(e) {
        console.error("Erro ao carregar o estado", e);
    }
}

function saveState() {
    try {
        localStorage.setItem('photo_os_state', JSON.stringify(appState));
    } catch (e) {
        alert("⚠️ Limite de espaço do LocalStorage atingido (~5MB)!");
    }
}

// Elementos carregados do DOM
const foldersContainer = document.getElementById('folders-container');
const galleryGrid = document.getElementById('gallery-grid');
const currentViewTitle = document.getElementById('current-view-title');
const fileInput = document.getElementById('file-input');
const uploadBox = document.getElementById('upload-box');
const trashTab = document.getElementById('trash-tab');
const trashCount = document.getElementById('trash-count');
const addFolderBtn = document.getElementById('add-folder-btn');
const lightbox = document.getElementById('lightbox');
const lightboxImg = document.getElementById('lightbox-img');
const closeBtn = document.getElementById('close-btn');

function init() {
    renderFolders();
    renderPhotos();
    setupEvents();
}

function renderFolders() {
    foldersContainer.innerHTML = '';
    appState.folders.forEach(folder => {
        const li = document.createElement('li');
        li.className = `folder-item \${appState.currentFolder === folder && !appState.isTrashView ? 'active' : ''}`;
        li.innerHTML = `
            <span class="folder-name">📁 \${folder}</span>
            <button class="delete-folder-btn" data-folder="\${folder}">&times;</button>
        `;

        li.addEventListener('click', (e) => {
            if(e.target.classList.contains('delete-folder-btn')) return;
            appState.currentFolder = folder;
            appState.isTrashView = false;
            renderFolders();
            renderPhotos();
        });

        li.querySelector('.delete-folder-btn').addEventListener('click', (e) => {
            e.stopPropagation();
            if(confirm(`Deseja apagar a pasta "\${folder}"?`)) {
                appState.photos.forEach(p => { if(p.folder === folder) p.isDeleted = true; });
                appState.folders = appState.folders.filter(f => f !== folder);
                appState.currentFolder = appState.folders[0] || '';
                saveState();
                renderFolders();
                renderPhotos();
            }
        });
        foldersContainer.appendChild(li);
    });

    if(appState.isTrashView) trashTab.classList.add('active');
    else trashTab.classList.remove('active');
    trashCount.textContent = `(\${appState.photos.filter(p => p.isDeleted).length})`;
}

function renderPhotos() {
    galleryGrid.innerHTML = '';
    if (appState.isTrashView) {
        currentViewTitle.textContent = "Lixeira";
        uploadBox.style.display = 'none';
    } else {
        currentViewTitle.textContent = appState.currentFolder || "Nenhuma pasta";
        uploadBox.style.display = appState.currentFolder ? 'block' : 'none';
    }

    const filtered = appState.photos.filter(p => appState.isTrashView ? p.isDeleted : (!p.isDeleted && p.folder === appState.currentFolder));

    if(filtered.length === 0) {
        galleryGrid.innerHTML = `<p style="color: #666; grid-column: 1/-1; text-align: center;">Nenhuma foto por aqui.</p>`;
        return;
    }

    filtered.forEach(photo => {
        const card = document.createElement('div');
        card.className = 'photo-card';
        card.innerHTML = `
            <div class="img-container"><img src="\${photo.src}"></div>
            <div class="photo-info">\${photo.name}</div>
            <div class="card-actions">
                \${appState.isTrashView ? 
                    `<button class="btn-action restore" onclick="restorePhoto('\${photo.id}')">Restaurar</button>
                     <button class="btn-action delete" onclick="permaDeletePhoto('\${photo.id}')">Excluir</button>` : 
                    `<button class="btn-action delete" onclick="deletePhoto('\${photo.id}')">Mover p/ Lixeira</button>`
                }
            </div>
        `;
        card.querySelector('.img-container').addEventListener('click', () => {
            lightboxImg.src = photo.src;
            lightbox.classList.add('active');
        });
        galleryGrid.appendChild(card);
    });
}

window.deletePhoto = function(id) {
    const photo = appState.photos.find(p => p.id === id);
    if(photo) { photo.isDeleted = true; saveState(); renderFolders(); renderPhotos(); }
}

window.restorePhoto = function(id) {
    const photo = appState.photos.find(p => p.id === id);
    if(photo) {
        photo.isDeleted = false;
        if(!appState.folders.includes(photo.folder)) {
            photo.folder = appState.folders[0] || 'Minhas Fotos';
            if(!appState.folders.includes(photo.folder) && photo.folder) appState.folders.push(photo.folder);
        }
        saveState(); renderFolders(); renderPhotos();
    }
}

window.permaDeletePhoto = function(id) {
    if(confirm("Apagar permanentemente?")) {
        appState.photos = appState.photos.filter(p => p.id !== id);
        saveState(); renderFolders(); renderPhotos();
    }
}

function readFileAsDataURL(file) {
    return new Promise((resolve) => {
        const reader = new FileReader();
        reader.onload = (evt) => resolve({ name: file.name, src: evt.target.result });
        reader.readAsDataURL(file);
    });
}

function setupEvents() {
    fileInput.addEventListener('change', async (e) => {
        const files = Array.from(e.target.files);
        if(!appState.currentFolder || files.length === 0) return;
        const imageFiles = files.filter(f => f.type.startsWith('image/'));
        
        const readResults = await Promise.all(imageFiles.map(f => readFileAsDataURL(f)));
        readResults.forEach((result, index) => {
            appState.photos.push({
                id: Date.now().toString() + index + Math.random().toString(36).substr(2, 5),
                name: result.name, src: result.src, folder: appState.currentFolder, isDeleted: false
            });
        });
        saveState(); renderFolders(); renderPhotos();
        fileInput.value = ''; 
    });

    addFolderBtn.addEventListener('click', () => {
        const name = prompt("Nome da nova pasta:");
        if(name && name.trim() !== "") {
            const clean = name.trim();
            if(appState.folders.includes(clean)) return alert("Já existe!");
            appState.folders.push(clean); appState.currentFolder = clean; appState.isTrashView = false;
            saveState(); renderFolders(); renderPhotos();
        }
    });

    trashTab.addEventListener('click', () => { appState.isTrashView = true; appState.currentFolder = ''; renderFolders(); renderPhotos(); });
    closeBtn.addEventListener('click', () => lightbox.classList.remove('active'));
    lightbox.addEventListener('click', (e) => { if (e.target !== lightboxImg && e.target !== closeBtn) lightbox.classList.remove('active'); });
    document.addEventListener('keydown', (e) => { if (e.key === 'Escape') lightbox.classList.remove('active'); });
}

init();
